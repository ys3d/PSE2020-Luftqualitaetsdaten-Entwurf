\makenoidxglossaries

\newglossaryentry{Leaflet}{
    name={Leaflet},
    description={Leaflet ist eine freie JavaScript-Bibliothek, mit der Web Anwendungen erstellt werden können. Die Bibliothek unterstützt somit die meisten Desktop- und Mobil-Browser.}
}

\newglossaryentry{Jsweet}{
    name={Jsweet},
    description={Jsweet ist ein freier Transpiler von Java hinzu TypeScript/JavaScript, welcher weitere Funktionen über Bibliotheken über sogenannte Candies hinzufügt}
}

\newglossaryentry{Angular-4-Candy}{
    name={Angular-4-Candy},
    description={Angular ist eine Plattform zum bauen von Mobil und Desktop Anwendungen. Angular 4 ist ein TypeScript-basiertes Front-End-Webapplikationsframework. Beim Angular-4-Candy handelt es sich um eine Jsweet Bibliothek welche dieses Framwork in Java einbindet.}
}
